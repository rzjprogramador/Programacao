namespace aula1 {
  class Program 
  {
  static void Main() 
  {
  // ENTRADA DINAMICA:
  Console.WriteLine("Digite o valor para operacao: ");
  var pergunta = Convert.ToInt32(Console.ReadLine()); 
  var valorPresente = 50.0;
  var resposta = pergunta - valorPresente; 
  
  Console.WriteLine($"O valor enviado foi {pergunta} do item que tem o estado fixo de {valorPresente} respondo que a operacao da pergunta que entrou com o estado presente é {resposta} .");  
  } 
  }
}
